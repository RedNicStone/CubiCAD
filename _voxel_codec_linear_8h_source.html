57std::shared_ptr<VoxelCodecLinear<Types...>> VoxelCodecLinear<Types...>::Allocate(AllocationHandler* handler,
58 VoxelRegion region) {
59 static_assert(std::is_class_v<fragment>, "Types must be a valid parameter pack for VoxelFragment");
60
61 std::shared_ptr<VoxelCodecLinear<Types...>> codec = std::make_shared<VoxelCodecLinear<Types...>>();
62
63 codec->allocationSize = region;
64 codec->size = region.x * region.y * region.z;
65 codec->allocationPtr = handler->makeAllocation(codec->size * fragment::getFragmentTotalSize());
66 codec->allocationHandler = handler;
67 codec->readWritePtr = codec->allocationPtr;
