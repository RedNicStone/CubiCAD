43template<typename Parameters>
44std::shared_ptr<MaterialLibrary<Parameters>> MaterialLibrary<Parameters>::create(const std::shared_ptr<Device>& pDevice,
45 const std::shared_ptr<Queue>& pTransferQueue,
46 const std::vector<std::shared_ptr<Queue>>&
47 accessingQueues,
48 size_t reservedMaterials) {
49 auto library = std::make_shared<MaterialLibrary<Parameters>>();
50 library->device = pDevice;
51 library->transferQueue = pTransferQueue;
52 library->materials = std::vector<std::shared_ptr<Material<Parameters>>>(reservedMaterials);
53
54 std::vector<glm::uint32> accessingQueueFamilyIndexes{};
55 accessingQueueFamilyIndexes.reserve(accessingQueues.size() + 1);
56 accessingQueueFamilyIndexes.push_back(pTransferQueue->getQueueFamilyIndex());
57 for (const auto& queue : accessingQueues) {
58 accessingQueueFamilyIndexes.push_back(queue->getQueueFamilyIndex());
59 }
60 library->materialBuffer = Buffer::create(pDevice, sizeof(Parameters) * reservedMaterials,
61 VMA_MEMORY_USAGE_GPU_ONLY, VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT | VK_BUFFER_USAGE_TRANSFER_DST_BIT,
62 accessingQueueFamilyIndexes);
63
64 return library;

67template<typename Parameters>
68std::shared_ptr<Material<Parameters>> MaterialLibrary<Parameters>::registerShader(Parameters parameters) {
69 if (materialCount + 1 < materials.size()) {
70 materials[materialCount + 1] = parameters;
71 } else {
72 materials.push_back(parameters);
73 materials.resize(materials.capacity());
74
75 materialBuffer = Buffer::create(device, sizeof(Parameters) * materials.size(),
76 VMA_MEMORY_USAGE_GPU_ONLY, VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT | VK_BUFFER_USAGE_TRANSFER_DST_BIT,
77 accessingQueueFamilyIndexes);
78 }
79 bufferOOD = true;
80}

82template<typename Parameters>
83void MaterialLibrary<Parameters>::pushParameters() {
84 if (bufferOOD) {
85 std::vector<Parameters> data(materials.size());
86 for (size_t i = 0; i < materials.size(); i++) {
87 data[i] = materials[i].getParameters();
88 }
89 materialBuffer->transferDataMapped(data.data());
90 }
