90 writeDescriptorSets[i].sType = VK_STRUCTURE_TYPE_WRITE_DESCRIPTOR_SET;
91 writeDescriptorSets[i].dstSet = handle;
92 writeDescriptorSets[i].dstBinding = binding[i];
