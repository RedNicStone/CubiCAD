66template<typename... Types>
67bool VoxelCodecSparseOctree<Types...>::enterNode() {
68 if (leaf) // a leaf can't have children
69 return false;
70
71 Node* currentNode = *nodeStack.top(); // acquire current node (static)
72 if (currentNode->validMask == 0x00) // check any node exists
73 return false;
74
75 auto first_node = static_cast<OctreePos>(getMSB(currentNode->validMask)); // get first node position
76 auto first_node_flag = static_cast<OctreePosFlag>((1 << first_node) >> 1); // get first node pos as flag
77 leaf = currentNode->leafMask & first_node_flag; // check if the node will be a leaf
78
79 uint_fast64_t pNextNode; // pointer to next node
80 if (currentNode->childPtr & 0x1) { // childPtr is a far pointer
81 pNextNode = (*allocation.getData(nodeStack.top().nodePtr + ((currentNode->childPtr >> 1) + 1) * 4)
82 + 1) * 4; // get value of far pointer
83 } else { // childPtr is a normal pointer
84 pNextNode = (currentNode->childPtr >> 1) + 1; // align with node pattern
85 }
86 pNextNode += nodeStack.top().nodePtr; // add current pointer to nextNode pointer
87 char* memPtr = allocation.getData(pNextNode); // get physical pointer to nextNode
88 nodeStack.push( { pNextNode, memPtr, first_node } ); // push node to stack
89}
